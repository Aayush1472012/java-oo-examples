\begin{frame}
  \frametitle{Hvor lang tid?}
  \begin{description}
  \item[Worst case] Hvor lang tid det tar på det meste
  \item[Best case] Hvor lang tid det tar på det minste
  \item[Average case] Hvor lang tid det tar i gjennomsnitt
  \end{description}
\end{frame}


\begin{frame}
\frametitle{Begreper}
  \begin{description}
  \item[Stabilitet] Er ting i rett rekkefølge etter sortering? Dette bryr vi oss ikke med når vi sorterer tall feks.
  \item[In-place] Trenger ikke ytterligere minne.
  \item[Adaptiv] Fungerer det bedre om listen allerede er noe sortert?
  \end{description}
\end{frame}

\begin{frame}
  \frametitle{Nye algoritmer?}
  \begin{itemize}
  \item Det forskes.
  \item library-sort kom på 2000-tallet
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Splitt og hersk}
  \begin{enumerate}
  \item \textbf{Divide} Hvis inputstørrelsen er mindre en angitt mål - løs problemet direkte. Ellers del opp.
  \item \textbf{Recursion} Rekursjer over de mindre delene
  \item \textbf{Conquer} Kombiner løsningene for smådelene
  \end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{Bubblesort}
  \begin{itemize}
  \item Enkel
  \item Lett å programmere
  \item Men håpløst treg
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Merge Sort}
  \begin{itemize}
  \item Typisk splitt og hersk
  \item \textbf{Divide} Er lengden på listen $1$ eller $0$ er listen sortert. Ellers del i to.
  \item \textbf{Recursion} Rekursjon over hver del vil gjøre susen!
  \item \textbf{Conquer} Sammenflett hver del.
  \end{itemize}
\end{frame}
